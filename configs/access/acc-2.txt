! ============================================================
! Device  : ACC-2  (labeled SW2 in source configs)
! Role    : Access Layer Switch — HR VLAN
! VLANs   : 30 (HR)
! Uplinks : G0/0 → CORE-1 (primary), G0/3 → CORE-2 (secondary)
! ============================================================

hostname ACC-2

!
! --- Spanning Tree ---
!
spanning-tree mode pvst
spanning-tree extend system-id

!
! --- Uplink Trunks ---
! Only VLAN 30 is allowed — HR traffic is the sole function
! of this switch. No other VLANs are present or needed.
!
interface GigabitEthernet0/0
 description PRIMARY-UPLINK-TO-CORE-1
 switchport trunk allowed vlan 30
 switchport trunk encapsulation dot1q
 switchport mode trunk
 negotiation auto

interface GigabitEthernet0/3
 description SECONDARY-UPLINK-TO-CORE-2
 switchport trunk allowed vlan 30
 switchport trunk encapsulation dot1q
 switchport mode trunk
 negotiation auto

!
! --- Access Ports ---
! All ports serve VLAN 30 (HR).
! G1/0 is an unassigned port; placed in VLAN 30 as a
! precaution. In production, unused ports should be
! disabled and assigned to a dedicated unused VLAN.
!

interface GigabitEthernet0/1
 description HR-PC1
 switchport access vlan 30
 switchport mode access
 negotiation auto
 spanning-tree portfast edge

interface GigabitEthernet0/2
 description HR-PC2
 switchport access vlan 30
 switchport mode access
 negotiation auto
 spanning-tree portfast edge

interface GigabitEthernet1/0
 description UNASSIGNED
 switchport access vlan 30
 switchport mode access
 negotiation auto
 shutdown
