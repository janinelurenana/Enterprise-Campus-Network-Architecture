! ============================================================
! Device  : ACC-1
! Role    : Access Layer Switch — Management & Guest VLANs
! VLANs   : 20 (Management), 40 (Guest)
! Uplinks : G0/0 → CORE-1 (primary), G1/1 → CORE-2 (secondary)
! ============================================================

hostname ACC-1

!
! --- Spanning Tree ---
! Rapid-PVST+ aligns with core STP mode.
! This switch is never the root; root roles are pinned to CORE-1/2.
!
spanning-tree mode pvst
spanning-tree extend system-id

!
! --- Uplink Trunks ---
! G0/0: Primary uplink to CORE-1.
!   Under normal STP operation this link forwards.
! G1/1: Secondary (cross) uplink to CORE-2.
!   Remains in blocking state under normal operation.
!   Rapid-PVST+ transitions to forwarding on primary uplink failure.
! Both trunks carry only VLANs 20 and 40 — no unnecessary
! broadcast domains leaked across the trunk.
!
interface GigabitEthernet0/0
 description PRIMARY-UPLINK-TO-CORE-1
 switchport trunk allowed vlan 20,40
 switchport trunk encapsulation dot1q
 switchport mode trunk
 negotiation auto

interface GigabitEthernet1/1
 description SECONDARY-UPLINK-TO-CORE-2
 switchport trunk allowed vlan 20,40
 switchport trunk encapsulation dot1q
 switchport mode trunk
 negotiation auto

!
! --- Access Ports ---
! PortFast edge: skips STP listening/learning on host ports.
!   Safe because these ports connect to end hosts, not switches.
! No BPDUGuard configured here — add in production to
! prevent accidental switch connections on access ports.
!

interface GigabitEthernet0/1
 description MGMT-PC1
 switchport access vlan 20
 switchport mode access
 negotiation auto
 spanning-tree portfast edge

interface GigabitEthernet0/2
 description GUEST-PC1
 switchport access vlan 40
 switchport mode access
 negotiation auto
 spanning-tree portfast edge

interface GigabitEthernet0/3
 description MGMT-PC2
 switchport access vlan 20
 switchport mode access
 negotiation auto
 spanning-tree portfast edge

interface GigabitEthernet1/0
 description GUEST-PC2
 switchport access vlan 40
 switchport mode access
 negotiation auto
 spanning-tree portfast edge
